## 5分钟快速安装
```bash

# 1. 安装 OpenClaw
npm install -g openclaw

# 2. 初始化配置
openclaw init

# 3. 启动 Gateway
openclaw gateway start

# 完成!
```

## 接入IM(三选一)

# 3. 编辑 ~/.openclaw/openclaw.json:
{
 "channels": {
 "telegram": {
 "enabled": true,
 "token": "你的Token"
 }

# 4. 重启
openclaw gateway restart

# 5. 安装插件
npm install -g @openclaw/channel-feishu

# 1. 安装 NapCat
git clone https://github.com/NapNeko/NapCatQQ.git
cd NapCatQQ && npm install

# 3. 启动
npm start

# 4. 在 OpenClaw 中配置
npm install -g @openclaw/channel-qq

# Gateway 管理
openclaw gateway status # 状态
openclaw gateway start # 启动
openclaw gateway stop # 停止
openclaw gateway restart # 重启
openclaw gateway logs # 日志

# 配置管理
openclaw config get # 查看配置
openclaw config schema # 配置结构

# Session 管理
openclaw sessions list # 会话列表
openclaw sessions send # 发送消息

## 故障排查

# 检查端口
netstat -tln | grep 18789

# 查看日志
tail -f ~/.openclaw/logs/gateway.log

# 检查配置
cat ~/.openclaw/openclaw.json | python3 -m json.tool

# 1. 检查 token
cat ~/.openclaw/openclaw.json | grep token

# 3. 查看 Telegram 日志
tail -f ~/.openclaw/logs/gateway.log | grep telegram

# 1. 检查 NapCat 是否运行
ps aux | grep napcat

# 2. 查看日志
tail -f NapCatQQ/logs/*.log

# 首次运行会显示二维码

# 快速备份
tar -czf backup-$(date +%Y%m%d).tar.gz \
 ~/.openclaw/openclaw.json \
 ~/.openclaw/workspace

# 恢复
tar -xzf backup-20260310.tar.gz -C ~/

# 查看帮助
openclaw --help
openclaw help

openclaw gateway logs

## 最佳实践
1. **定期更新**
 npm update -g openclaw

2. **定期备份**
 # 每周备份一次
 crontab -e
 # 添加:0 0 * * 0 tar -czf ...

3. **定期清理**
 # 清理7天前的日志
 find ~/.openclaw/logs -name "*.log" -mtime +7 -delete

4. **使用环境变量**
 # 在 ~/.bashrc 中
 export OPENCLAW_TELEGRAM_TOKEN="your-token"
